**Module design:**

```typescript
import type { FindingClaimClassification, ClaimClassification } from "./claim-classifier.ts";

/** Minimum word count after stripping external sentences. Below this, suppress entirely. */
const MIN_WORDS_AFTER_REWRITE = 10;

/** Footnote appended to rewritten findings */
const REWRITE_FOOTNOTE = "\n\nℹ️ Some claims removed (unverifiable)";

/** Input: a finding with title and optional claim classification */
export type FilterableFinding = {
  title: string;
  commentId: number;
  severity: string;
  category: string;
  filePath: string;
  claimClassification?: FindingClaimClassification;
  [key: string]: unknown;
};

/** Action taken by the filter */
export type FilterAction = "rewritten" | "suppressed";

/** A finding that was suppressed or rewritten, with audit data */
export type FilteredFindingRecord = {
  commentId: number;
  originalTitle: string;
  action: FilterAction;
  rewrittenTitle?: string;
  reason: string;
  classificationEvidence: string[];
};

/** Result of filtering a batch of findings */
export type FilterResult<T extends FilterableFinding> = {
  /** Findings that survived (may have rewritten titles) */
  findings: T[];
  /** Records of all suppressed/rewritten findings for logging */
  filtered: FilteredFindingRecord[];
  /** Count of rewritten findings */
  rewriteCount: number;
  /** Count of suppressed findings */
  suppressionCount: number;
};

export function filterExternalClaims<T extends FilterableFinding>(
  findings: T[],
  logger?: { info: (obj: Record<string, unknown>, msg: string) => void },
): FilterResult<T>;

export function formatSuppressedFindingsSection(
  filtered: FilteredFindingRecord[],
): string;
```

**filterExternalClaims logic:**

1. For each finding, check `finding.claimClassification?.summaryLabel`
2. If undefined or `"primarily-diff-grounded"` => pass through to output findings unchanged
3. If `"primarily-external"` => suppress: do NOT include in output findings, add to filtered records with action="suppressed", reason built from external claim evidence strings
4. If `"mixed"` => rewrite:
   a. Get the claims array from claimClassification
   b. Collect sentences where `claim.label !== "external-knowledge"` (keep diff-grounded AND inferential)
   c. Join kept sentences with space
   d. Count words in the result. If < MIN_WORDS_AFTER_REWRITE, treat as suppression instead (too short to be useful)
   e. Otherwise, append REWRITE_FOOTNOTE to the joined text
   f. Return finding with `{ ...finding, title: rewrittenTitle }` in output findings
   g. Add to filtered records with action="rewritten", originalTitle, rewrittenTitle
5. For each suppressed/rewritten finding, if logger provided, log at info level with: `{ commentId, originalTitle, action, rewrittenTitle?, reason, summaryLabel }`

**formatSuppressedFindingsSection logic:**

1. Filter for action="suppressed" only
2. If empty, return "" (no section at all)
3. Build a collapsed `<details>` section:
```html
<details>
<summary>Filtered findings (N findings removed — unverifiable claims)</summary>

- **Finding title** — reason
- **Finding title** — reason

</details>
```
Use the `originalTitle` (truncated to 80 chars if longer) and `reason` from each FilteredFindingRecord.

```typescript
// Output filtering: rewrite mixed findings, suppress primarily-external findings (FILT-01, FILT-02)
const filterResult = filterExternalClaims(
  processedFindings as FilterableFinding[],
  logger,
);

// Apply filter results back to processedFindings
if (filterResult.suppressionCount > 0 || filterResult.rewriteCount > 0) {
  // Build lookup maps from filter result
  const suppressedIds = new Set(
    filterResult.filtered
      .filter(r => r.action === "suppressed")
      .map(r => r.commentId),
  );
  const rewriteMap = new Map(
    filterResult.filtered
      .filter(r => r.action === "rewritten")
      .map(r => [r.commentId, r.rewrittenTitle!]),
  );

  processedFindings = processedFindings.map(f => {
    if (suppressedIds.has(f.commentId)) {
      return { ...f, suppressed: true, filterAction: "suppressed" as const, originalTitle: f.title };
    }
    const rewrittenTitle = rewriteMap.get(f.commentId);
    if (rewrittenTitle) {
      return { ...f, title: rewrittenTitle, filterAction: "rewritten" as const, originalTitle: f.title };
    }
    return f;
  });

  // Log filter summary (FILT-03)
  logger.info(
    {
      ...baseLog,
      rewriteCount: filterResult.rewriteCount,
      suppressionCount: filterResult.suppressionCount,
      filteredFindings: filterResult.filtered.map(r => ({
        commentId: r.commentId,
        action: r.action,
        originalTitle: r.originalTitle.slice(0, 100),
        reason: r.reason,
      })),
    },
    "Output filter applied: external knowledge claims filtered",
  );
}
```
